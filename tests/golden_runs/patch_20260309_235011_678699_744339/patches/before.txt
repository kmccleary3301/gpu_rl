from __future__ import annotations

import argparse
import json
from pathlib import Path
import sys

import torch

HERE = Path(__file__).resolve().parent
if str(HERE) not in sys.path:
    sys.path.insert(0, str(HERE))

from triton_attention_score_shared import HIDDEN_K, HIDDEN_Q, VISIBLE_K, VISIBLE_Q, reference_attention_tensor


def _benchmark_inputs(n_q: int, n_k: int, head_dim: int) -> tuple[torch.Tensor, torch.Tensor]:
    q_base = torch.arange(n_q * head_dim, device="cuda", dtype=torch.float32)
    k_base = torch.arange(n_k * head_dim, device="cuda", dtype=torch.float32)
    q = ((q_base.reshape(n_q, head_dim) * 13) % 257) / 257.0
    k = ((k_base.reshape(n_k, head_dim) * 29) % 251) / 251.0
    return q, k


def _weak_attention_scores(q: torch.Tensor, k: torch.Tensor) -> torch.Tensor:
    # Intentionally weak row-wise formulation used as the reformulation baseline.
    outputs: list[torch.Tensor] = []
    for row in range(int(q.shape[0])):
        row_scores = reference_attention_tensor(q[row : row + 1], k, causal=False).squeeze(0)
        row_scores = torch.where(
            torch.arange(k.shape[0], device=q.device) <= row,
            row_scores,
            torch.full_like(row_scores, -1.0e9),
        )
        outputs.append(row_scores)
    return torch.stack(outputs, dim=0)


def _run_benchmark(repeats: int) -> None:
    q, k = _benchmark_inputs(n_q=512, n_k=512, head_dim=64)
    for _ in range(repeats):
        # Keep the baseline materially weaker than the tiled candidate under
        # end-to-end subprocess timing by doing several row-wise passes.
        for _baseline_pass in range(4):
            out = _weak_attention_scores(q, k)
        torch.cuda.synchronize()
    _ = out


def _serialize(q: torch.Tensor, k: torch.Tensor) -> list[list[float]]:
    scores = _weak_attention_scores(q.cuda(), k.cuda()).cpu()
    return [[round(float(value), 6) for value in row] for row in scores.tolist()]


def main() -> None:
    parser = argparse.ArgumentParser()
    parser.add_argument("--benchmark-repeats", type=int, default=20)
    args = parser.parse_args()

    _run_benchmark(args.benchmark_repeats)
    print(
        json.dumps(
            {
                "visible_attention_scores": _serialize(VISIBLE_Q, VISIBLE_K),
                "hidden_attention_scores": _serialize(HIDDEN_Q, HIDDEN_K),
                "reformulation_summary": {
                    "baseline_ref": "workloads/reference/triton_attention_score_weak_baseline.py",
                    "optimized_ref": "workloads/reference/triton_attention_score_reformulate_candidate.py",
                    "strategy_change": "replace_naive_python_loops_with_tiled_triton_kernel",
                },
            },
            sort_keys=True,
        )
    )


if __name__ == "__main__":
    main()
